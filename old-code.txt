	public static void saveExpensesInFile(ArrayList<Expense> expenses, String fileName)
	{
        try {
              
        	var fos = new FileOutputStream(fileName);
			var oos = new ObjectOutputStream(fos);
			oos.writeObject(expenses);
        }
        catch(Exception e) {
        	System.out.println("Error: " + e.getMessage());
        }
      
	}
	
	public static ArrayList<Expense> GetExpensesFromFile(String fileName)
	{
		try{
			var fis = new FileInputStream(fileName);
			var ois = new ObjectInputStream(fis);
			var items=  (ArrayList<Expense>)ois.readObject();
			return items;
		} 
		catch(Exception e) {
        	System.out.println("Error: " + e.getMessage());
        }
		return null;
	}
